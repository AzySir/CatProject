```
{
  "name": "nodeexample",
  "version": "1.0.0",
  "description": "",
  "main": "index.js",
  "dependencies": {
    "axios": "^0.19.0",

  },
  "devDependencies": {
    "mocha": "^6.1.4"
  },
  "scripts": {
    "start": "node index.js",
    "test": "mocha"
  },
  "repository": {
    "type": "git",
    "url": "git+https://github.com/AzySir/CatProject.git"
  },
  "author": "",
  "license": "ISC",
  "bugs": {
    "url": "https://github.com/AzySir/CatProject/issues"
  },
  "homepage": "https://github.com/AzySir/CatProject#readme"
}
```
2) To start project ```npm start```
3) To test ``` npm test ```
